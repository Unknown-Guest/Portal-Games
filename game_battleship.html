function sendBattleshipInvite(){ const code = document.getElementById('bs-ol-room')?.textContent?.trim() || document.getElementById('bs-ol-code')?.value?.trim().toUpperCase(); if(!code){ document.getElementById('bs-ol-status').textContent = 'Create or join a room first.'; return; } const username = prompt('Enter username to invite:'); if(!username) return; const invites = JSON.parse(localStorage.getItem('gameInvites') || '[]'); invites.push({ to: username.trim(), from: localStorage.getItem('loggedInUser') || 'Player', game: 'Battleship', roomCode: code, link: `game_battleship.html?invite=${code}`, time: Date.now() }); localStorage.setItem('gameInvites', JSON.stringify(invites)); document.getElementById('bs-ol-status').textContent = 'Invite sent to ' + username; }
